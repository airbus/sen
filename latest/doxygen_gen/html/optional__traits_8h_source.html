48template <typename T>
49inline void OptionalTraitsBase<T>::write(OutputStream& out, T val)
50{
51 auto hasValue = val.has_value();
52 out.writeBool(hasValue);
53 if (hasValue)
54 {
55 SerializationTraits<typename T::value_type>::write(out, *val);
56 }
57}

59template <typename T>
60inline void OptionalTraitsBase<T>::read(InputStream& in, T& val)
61{
62 bool hasValue = false;
63 in.readBool(hasValue);
64
65 if (hasValue)
66 {
67 typename T::value_type content;
68 SerializationTraits<typename T::value_type>::read(in, content);
69 val = T(content);
70 }
71 else
72 {
73 val = T(std::nullopt);
74 }
75}

77template <typename T>
78inline uint32_t OptionalTraitsBase<T>::serializedSize(T val) noexcept
79{
80 auto hasValue = val.has_value();
81 auto result = impl::getSerializedSize(hasValue);
82
83 if (hasValue)
84 {
85 result += SerializationTraits<typename T::value_type>::serializedSize(*val);
86 }
87
88 return result;
89}

91template <typename T>
92inline void OptionalTraitsBase<T>::valueToVariant(const T& val, Var& var)
93{
94 if (val.has_value())
95 {
96 VariantTraits<typename T::value_type>::valueToVariant(*val, var);
97 }
98 else
99 {
100 var = {};
101 }
102}

104template <typename T>
105inline void OptionalTraitsBase<T>::variantToValue(const Var& var, T& val)
106{
107 // the var might be empty or hold an empty map
108 if (var.isEmpty() || (var.holds<VarMap>() && var.get<VarMap>().empty()))
109 {
110 val = T(std::nullopt);
111 }
112 else
113 {
114 typename T::value_type content;
115 VariantTraits<typename T::value_type>::variantToValue(var, content);
116 val = T(content);
117 }
