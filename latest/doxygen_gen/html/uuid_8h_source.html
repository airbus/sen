249inline UuidVariant Uuid::getVariant() const noexcept
250{
251 if ((bytes_[8U] & 0x80) == 0x00) // NOLINT
252 {
253 return UuidVariant::ncs;
254 }
255 else if ((bytes_[8U] & 0xC0) == 0x80) // NOLINT
256 {
257 return UuidVariant::rfc;
258 }
259 else if ((bytes_[8U] & 0xE0) == 0xC0) // NOLINT NOSONAR
260 {
261 return UuidVariant::microsoft;
262 }
263
264 return UuidVariant::reserved;
265}

280inline UuidVersion Uuid::getVersion() const noexcept
281{

314inline uint64_t Uuid::getHash() const noexcept
315{
316 return *(reinterpret_cast<const uint64_t*>(bytes_.data())) ^ // NOLINT NOSONAR
317 *(reinterpret_cast<const uint64_t*>(bytes_.data()) + sizeof(uint64_t)); // NOLINT NOSONAR
318}

320inline uint32_t Uuid::getHash32() const noexcept
321{
322 const auto hash64 = getHash();
323 return static_cast<uint32_t>(hash64 ^ (hash64 >> 32U)); // NOLINT NOSONAR
324}

326inline bool Uuid::operator==(const Uuid& rhs) const noexcept
327{
328 __m128i x = _mm_load_si128(constDataCast(bytes_.data()));
329 __m128i y = _mm_load_si128(constDataCast(rhs.bytes_.data()));
330 __m128i neq = _mm_xor_si128(x, y);
331 return _mm_test_all_zeros(neq, neq);
332}

334inline bool Uuid::operator!=(const Uuid& rhs) const noexcept { return !(*this == rhs); }

336inline bool Uuid::operator<(const Uuid& rhs) const noexcept
337{
338 // There are no trivial 128-bits comparisons in SSE/AVX
339 // It's faster to compare two uint64_t
340 const auto* x = reinterpret_cast<const uint64_t*>(bytes_.data()); // NOLINT NOSONAR
341 const auto* y = reinterpret_cast<const uint64_t*>(rhs.bytes_.data()); // NOLINT NOSONAR
342 return *x < *y || (*x == *y && *(x + 1U) < *(y + 1U)); // NOLINT NOSONAR
343}

349inline Uuid UuidRandomGenerator::operator()()
350{
351 // The two masks set the uuid version (4) and variant (1)
352 const __m128i and_mask = _mm_set_epi64x(0xffffffffffffff3fULL, 0xff0fffffffffffffULL); // NOLINT
353 const __m128i or_mask = _mm_set_epi64x(0x0000000000000080ULL, 0x0040000000000000ULL); // NOLINT
354 __m128i n = _mm_set_epi64x(distribution_(generator_), distribution_(generator_)); // NOLINT
355 __m128i uuid = _mm_or_si128(_mm_and_si128(n, and_mask), or_mask);
356
357 return Uuid(uuid);
358}
