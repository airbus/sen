template<class... Ts>
struct sen::Overloaded< Ts >

Helper type for std::variant lambda visitors.

Example:

std::visit(
sen::Overloaded {[&](uint8_t val) { ... },
[&](int16_t val) { ... },
[&](uint16_t val) { ... },
[&](int32_t val) { ... },
[&](uint32_t val) { ... },
[&](int64_t val) { ... },
[&](uint64_t val) { ... },
[&](float32_t val) { ... },
[&](float64_t val) { ... }},
variant);
};
