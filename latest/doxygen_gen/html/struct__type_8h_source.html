155{
156 u32 operator()(const StructSpec& spec) const noexcept
157 {
158 auto result = hashCombine(
159 hashSeed, spec.name, spec.qualifiedName, spec.parent ? spec.parent.value()->getHash() : nonPresentTypeHash);
160
161 for (const auto& field: spec.fields)
162 {
163 result = hashCombine(result, field);
164 }
