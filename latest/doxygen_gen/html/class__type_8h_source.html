237template <>
238struct impl::hash<ClassSpec>
239{
240 u32 operator()(const ClassSpec& spec) const noexcept
241 {
242 auto result = hashCombine(hashSeed, spec.name, spec.qualifiedName);
243
244 for (const auto& prop: spec.properties)
245 {
246 result = hashCombine(result, prop);
247 }
248
249 for (const auto& method: spec.methods)
250 {
251 result = hashCombine(result, method);
252 }
253
254 for (const auto& event: spec.events)
255 {
256 result = hashCombine(result, event);
257 }
258
259 result = hashCombine(
260 result, spec.constructor.has_value() ? impl::hash<MethodSpec>()(spec.constructor.value()) : nonPresentTypeHash);
261
262 for (const auto& parent: spec.parents)
263 {
264 result = hashCombine(result, parent->getHash());
265 }
266
267 return result;
268 }
269};
