32struct VariantField final
33{
34 VariantField(uint32_t key, std::string description, ConstTypeHandle<> type)
35 : key(key), description(std::move(description)), type(std::move(type))
36 {
37 SEN_ENSURE(!this->type->isVoidType() && "Variants cannot have void fields.");
38 }
39
40 // Comparison operators
41 friend bool operator==(const VariantField& lhs, const VariantField& rhs) noexcept
42 {
43 return lhs.key == rhs.key && lhs.description == rhs.description && lhs.type == rhs.type;
44 }
45
46 friend bool operator!=(const VariantField& lhs, const VariantField& rhs) noexcept { return !(lhs == rhs); }

128struct impl::hash<VariantSpec>
129{
130 u32 operator()(const VariantSpec& spec) const noexcept
131 {
132 auto result = hashCombine(hashSeed, spec.name, spec.qualifiedName);
