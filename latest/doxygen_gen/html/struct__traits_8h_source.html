86inline const VarMap& StructTraitsBase::getMap(const Var& var, std::string_view structType)
87{
88 static VarMap emptyMap = {};
89
90 if (var.isEmpty())
91 {
92 return emptyMap;
93 }
94
95 if (var.holds<VarMap>())
96 {
97 return var.get<VarMap>();
98 }
99
100 std::string err;
101 err.append("struct ");
102 err.append(structType);
103 err.append(": variant does not hold a variant map (");
104 err.append(var.getCopyAs<std::string>());
105 err.append(")");
106 throw std::runtime_error(err);
107}

109template <typename T>
110inline void StructTraitsBase::extractField(const VarMap& map,
111 T& val,
112 std::string_view fieldName,
113 std::string_view fieldType,
114 std::string_view structType)
115{
116 std::ignore = fieldType;
117 std::ignore = structType;
118
119 if (const auto itr = map.find(fieldName); itr != map.end())
120 {
121 ::sen::VariantTraits<T>::variantToValue(itr->second, val);
122 }
123}
