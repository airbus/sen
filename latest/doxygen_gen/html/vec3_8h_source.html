327template <typename T>
328inline T Vec3<T>::normalize() noexcept
329{
330 T norm = Vec3::length();
331 if (norm > 0.0)
332 {
333 T inv = 1.0f / norm;
334 v_[0] *= inv;
335 v_[1] *= inv;
336 v_[2] *= inv;
337 }
338 return (norm);
