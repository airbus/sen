34struct QuantitySpec final
35{
36 QuantitySpec(std::string name,
37 std::string qualifiedName,
38 std::string description,
39 ConstTypeHandle<NumericType> elementType,
40 std::optional<const Unit*> unit = std::nullopt,
41 std::optional<float64_t> minValue = std::nullopt,
42 std::optional<float64_t> maxValue = std::nullopt)
43 : name(std::move(name))
44 , qualifiedName(std::move(qualifiedName))
45 , description(std::move(description))
46 , elementType(std::move(elementType))
47 , unit(unit)
48 , minValue(minValue)
49 , maxValue(maxValue)
50 {
51 SEN_ASSERT(!unit || unit.value() != nullptr);
52 }
53
54 // Comparison operators.
55 friend bool operator==(const QuantitySpec& lhs, const QuantitySpec& rhs) noexcept
56 {
57 return lhs.description == rhs.description &&
58 (lhs.unit ? (rhs.unit && *lhs.unit == *rhs.unit) : rhs.unit == nullptr) &&
59 lhs.elementType == rhs.elementType && lhs.maxValue == rhs.maxValue && lhs.minValue == rhs.minValue;
60 }
61
62 friend bool operator!=(const QuantitySpec& lhs, const QuantitySpec& rhs) noexcept { return !(lhs == rhs); }
63
64public:
65 std::string name; // NOLINT(misc-non-private-member-variables-in-classes)
66 std::string qualifiedName; // NOLINT(misc-non-private-member-variables-in-classes)
67 std::string description; // NOLINT(misc-non-private-member-variables-in-classes)
68 ConstTypeHandle<NumericType> elementType; // NOLINT(misc-non-private-member-variables-in-classes)
69 std::optional<const Unit*> unit; // NOLINT(misc-non-private-member-variables-in-classes)
70 std::optional<float64_t> minValue; // NOLINT(misc-non-private-member-variables-in-classes)
71 std::optional<float64_t> maxValue; // NOLINT(misc-non-private-member-variables-in-classes)
72};
