103template <>
104struct VariantTraits<Duration>
105{
106 static void valueToVariant(Duration val, Var& var) { var = val; }
107
108 static void variantToValue(const Var& var, Duration& val)
109 {
110 // try to convert from string, maybe with custom unit
111 if (var.holds<std::string>())
112 {
113 auto result = QuantityTraits<Duration>::unit().fromString(var.get<std::string>());
114 if (result.isOk())
115 {
116 val = Duration(result.getValue());
117 return;
118 }
119 }
120
121 val = var.getCopyAs<Duration>();
122 }
123};
